  +def resolve_custom_domain(
  +    host: str,
  +    provider_domains: Mapping[DomainProvider, Collection[str]],
  +) -> DomainMatch:
  +    """Resolve which provider owns a host by longest matching configured zone.
  +
  +    The domain authority should be selected by who actually holds the zone, not
  +    by a user preference. When multiple providers claim the same longest suffix,
  +    the result is treated as ambiguous and must be resolved explicitly.
  +    """
  +
  +    normalized = host.strip().strip(".").lower()
  +    labels = [label for label in normalized.split(".") if label]
  +    if len(labels) < 2:
  +        raise ValueError(f"invalid host: {host}")
  +
  +    matches: list[tuple[int, DomainProvider, str]] = []
  +    for provider, domains in provider_domains.items():
  +        for domain in domains:
  +            candidate = domain.strip().strip(".").lower()
  +            if not candidate:
  +                continue
  +            if normalized == candidate or normalized.endswith(f".{candidate}"):
  +                matches.append((len(candidate), provider, candidate))
  +
  +    if not matches:
  +        raise ValueError(
  +            f"Could not determine who holds {normalized}. Configure the owning zone in Cloudflare or sesame/Porkbun, or override the domain provider explicitly."
  +        )
  +
  +    matches.sort(key=lambda item: item[0], reverse=True)
  +    best_len = matches[0][0]
  +    best = [entry for entry in matches if entry[0] == best_len]
  +    unique = {(provider, zone) for _, provider, zone in best}
  +    if len(unique) != 1:
  +        providers = ", ".join(sorted(provider.value for provider, _ in unique))
  +        raise ValueError(f"Ambiguous domain ownership for {normalized}; matching zones found in: {providers}")
  +
  +    _, provider, zone = best[0]
  +    subdomain = None if normalized == zone else normalized[: -(len(zone) + 1)]
  +    return DomainMatch(provider=provider, zone=zone, subdomain=subdomain)

  +#### split_custom_domain
  +Splits a full `host` (e.g., `app.dev.example.com`) into its `zone` and `subdomain` parts based on a list of `owned_domains`, preferring the longest match.
  +```python
  +def split_custom_domain(host: str, owned_domains: Collection[str]) -> tuple[str, str | None]:
  +    """Splits a full custom domain host into its zone and subdomain parts.\n\n    Example: host='app.dev.example.com', owned_domains=['example.com', 'dev.example.com']\n    Returns: ('dev.example.com', 'app')\n\n    It prefers the longest matching owned domain suffix.
  +    """
  +    longest_zone = ""
  +    for domain in owned_domains:
  +        if host.endswith(domain) and len(domain) > len(longest_zone):
  +            longest_zone = domain
  +
  +    if not longest_zone:
  +        raise ValueError(f"Host {host} does not match any owned domains.")
  +
  +    # If host is exactly the zone, there is no subdomain.
  +    if host == longest_zone:
  +        return longest_zone, None
  +
  +    # Subdomain is the part before the longest_zone, excluding the dot
  +    subdomain = host[: -len(longest_zone)].rstrip(".")
  +    return longest_zone, subdomain or None # subdomain can be empty if it's just the zone.
  +```

  +#### diagnose_den
  +Runs a series of diagnostic checks on the Den environment to verify health and readiness, stopping on the first failure unless specified.
  +```python
  +@mcp.tool
  +def diagnose_den(include_docker_build: bool = False) -> dict[str, Any]:
  +    """Run diagnostic checks on den tooling and setup.\n\n    This is a quick way to verify that the CLI and its dependencies are working.
  +    """
  +    steps: list[StepResult] = []
  +
  +    # Define the diagnostic steps
  +    diagnostic_steps = [
  +        ("mypy", ["mypy", PROJECT_DIR], PROJECT_DIR, 300, "`mypy` checks failed.", ["Fix type errors in the project."]),
  +        ("property_tests", ["pytest", "-m", "properties", str(PROJECT_DIR)], PROJECT_DIR, 300, "Property tests failed.", ["Inspect test failures and fix."]),
  +        ("den_smoke", ["bash", "-lc", "den list"], PROJECT_DIR, 30, "Den smoke test failed (list command).", ["Check `den list` manually."]),
  +    ]
  +
  +    if include_docker_build:
  +        diagnostic_steps.append(
  +            ("den_full", ["bash", "-lc", "den build-image"], PROJECT_DIR, 900, "Den full build failed.", ["Inspect logs, fix build errors."])
  +        )
  +
  +    for step_name, command, cwd, timeout_s, error_msg, remediation in diagnostic_steps:
  +        step = _run_step(step_name, command, cwd=cwd, timeout_s=timeout_s)
  +        steps.append(step)
  +        if not step["ok"]:
  +            return _result(
  +                "diagnose_den",
  +                False,
  +                steps=steps,
  +                error=_build_error(step, error_msg, remediation),
  +            )
  +
  +    return _result(
  +        "diagnose_den",
  +        True,
  +        steps=steps,
  +        data={"message": "All diagnostic checks passed."},
  +        next_steps=["Your den environment is ready."],
  +    )
  +```

  +#### redeploy
  +Triggers a redeployment of a Den `name` by creating a checkpoint and then restoring it.
  +```python
  +@app.command()
  +def redeploy(
  +    name: Annotated[str, typer.Argument(help="Name of the den")],
  +) -> None:
  +    """Create a checkpoint and restore it, forcing a redeploy.\n\n    This is useful for applying configuration changes or restarting a stuck den.
  +    """
  +    _ensure_project_dir()
  +    den_name = normalize_den_name(name)
  +    typer.secho(f"==> Redeploying den {den_name}...", fg=typer.colors.CYAN)
  +
  +    nonce = str(time.time_ns())
  +    comment = make_sprite_redeploy_comment(den_name, nonce)
  +
  +    # Create a new checkpoint
  +    _run_checked(
  +        sprite_checkpoint_create_command(den_name, comment),
  +        cwd=PROJECT_DIR,
  +        error_hint="Checkpoint creation failed.",
  +    )
  +
  +    # Find the version ID of the newly created checkpoint
  +    # First try API, then fallback to list output
  +    version_id = None
  +    proc_api = _run(sprite_command("api", "/checkpoints", sprite_name=den_name), capture_output=True)
  +    if proc_api.returncode == 0:
  +        version_id = find_checkpoint_version_in_api_output(proc_api.stdout, comment)
  +    if version_id is None:
  +        proc_list = _run(sprite_command("checkpoint", "list", sprite_name=den_name), capture_output=True)
  +        if proc_list.returncode == 0:
  +            version_id = find_checkpoint_version_in_list_output(proc_list.stdout, comment)
  +
  +    if not version_id:
  +        raise CommandError(f"Failed to find newly created checkpoint for {den_name}")
  +
  +    # Restore to the new checkpoint
  +    _run_checked(
  +        sprite_restore_command(den_name, version_id),
  +        cwd=PROJECT_DIR,
  +        error_hint=f"Checkpoint restore to {version_id} failed.",
  +    )
  +
  +    typer.secho(f"Den {den_name} redeployed successfully to version {version_id}.", fg=typer.colors.GREEN)
  +```

  +# Den CLI Technical Documentation
  +
  +## 1. Introduction
  +
  +The `den_cli` project is a Python-based command-line interface (CLI) and Multi-Agent Control Protocol (MCP) server designed to manage remote development environments referred to as "dens". These environments are orchestrated using the "Sprite" platform for hosting and potentially leverage "sesame" for custom domain management with providers like Porkbun. The tool aims to simplify the lifecycle management of these development environments, from initial setup and provisioning to ongoing operation and diagnostics.
  +
  +## 2. Architecture Overview
  +
  +The `den_cli` codebase is structured into three primary components, promoting a clear separation of concerns:
  +
  +1.  **Core Logic (`den_cli.core`)**: This module contains the foundational business logic, including utility functions for name normalization, construction of external command arguments, detection of project-specific markers, and inference of optimal environment configurations. It operates on data structures like `DenPeer`, `ProjectMarkers`, and `InferredDenSetup`.
  +2.  **CLI Interface (`den_cli.cli`)**: This is the user-facing component, built using the `Typer` library. It translates user commands into specific actions, orchestrates workflows by calling functions from the `core` module, and interacts with external tools (e.g., `sprite`, `sesame`) via `subprocess` execution. It defines various commands like `setup`, `exec`, `logs`, and `redeploy`.
  +3.  **MCP Server (`den_cli.mcp_server`)**: An API-driven server built with `fastmcp`, designed for automated and programmatic control of den environments. It exposes tools like `provision_den`, `operate_den`, and `diagnose_den`, providing structured JSON responses, detailed step-by-step results, and robust error handling suitable for integration with other automated systems.
  +
  +## 3. Key Components
  +
  +### 3.1. `den_cli.core` (Core Logic)
  +
  +The `den_cli.core` module provides the fundamental building blocks and business logic for interacting with den environments.
  +
  +*   **Data Structures**:
  +    *   `DenPeer`: Represents a minimal view of a den peer, holding its hostname, IP, and online status.
  +    *   `ProjectMarkers`: Captures filesystem signals (e.g., `package.json`, `pyproject.toml`, `Dockerfile`) to inform setup decisions.
  +    *   `InferredDenSetup`: Stores the derived setup plan, including the chosen backend (`nix` or `guix`), packages, and environment variables, before Dhall rendering.
  +    *   `InferredRunCommand`: Represents the inferred command for live development within a sprite.
  +
  +*   **Naming & ID Normalization**:
  +    *   The `normalize_den_name` function ensures that human-friendly den names adhere to Sprite's naming conventions, prefixing them with "den-" if necessary.
  +    *   Conversely, `short_den_name` removes this "den-" prefix for display purposes.
  +
  +*   **Sprite CLI Command Builders**:
  +    These functions generate `sprite` CLI commands, abstracting argument construction and ensuring consistent use of the preferred Sprite organization (`sprite_org`) and normalized den names.
  +    *   `sprite_command`: The base builder for any `sprite` command.
  +    *   `sprite_exec_command`: Constructs commands for non-interactive execution within a den.
  +    *   `sprite_tty_exec_command`: Builds commands for interactive TTY sessions.
  +    *   `sprite_use_command`: Generates commands to bind the current directory to a sprite.
  +    *   `sprite_logs_command`: Creates commands for listing sessions or attaching to logs.
  +    *   `sprite_checkpoint_create_command`: Builds commands to create a checkpoint.
  +    *   `sprite_restore_command`: Constructs commands to restore a checkpoint by ID.
  +    *   `make_sprite_redeploy_comment`: Creates a unique comment for checkpoints during redeploy operations, allowing them to be easily found.
  +
  +*   **Project Inference and Dhall Rendering**:
  +    *   `detect_project_markers`: Scans a given repository path for the presence of various project configuration files (e.g., `package.json`, `Cargo.toml`, `Dockerfile`, `flake.nix`, `guix/manifest.scm`).
  +    *   `infer_den_setup`: Uses the detected `ProjectMarkers` to determine the most suitable backend (`nix` or `guix`), default packages (`_infer_nix_packages`, `_infer_guix_packages`), and environment variables for the den, providing `reasons` for its decisions.
  +    *   `_infer_backend`: Internal helper that applies heuristics to choose between `nix` and `guix` based on markers.
  +    *   `render_den_dhall`: Takes the `InferredDenSetup` and generates a Dhall configuration string, which serves as a declarative blueprint for the den environment.
  +
  +*   **Domain and URL Utilities**:
  +    *   `parse_sprite_url`: Extracts the public URL from `sprite url` command output.
  +    *   `split_custom_domain`: Parses a custom domain into its subdomain and zone components, considering a list of owned domains.
  +    *   `discover_porkbun_domains_from_sesame_config`: Attempts to find owned Porkbun domains from local `sesame` configuration files.
  +    *   `resolve_sesame_command`: Determines the executable path for the `sesame` CLI tool.
  +
  +### 3.2. `den_cli.cli` (Command Line Interface)
  +
  +The `den_cli.cli` module defines the primary user interaction point, built as a `Typer` application.
  +
  +*   **Application Setup**:
  +    *   `app`: The main `Typer` application instance, configured with `APP_NAME` and help text.
  +    *   `PROJECT_DIR`, `DHALL_DIR`, `SECRETS_FILE`: Global constants defining important directory paths and file locations.
  +
  +*   **Subprocess Execution Helpers**:
  +    *   `_run`: A wrapper around `subprocess.run` for executing external commands, with options for `cwd`, `capture_output`, and `input_text`.
  +    *   `_run_checked`: Extends `_run` by performing error checking (`proc.returncode != 0`) and raising a `CommandError` if the command fails, providing error hints and output.
  +    *   `_command_exists`: Checks if a specified command exists in the system's PATH.
  +
  +*   **Den Management Utilities**:
  +    *   `_ensure_project_dir`: Verifies the existence of the expected `PROJECT_DIR`.
  +    *   `_list_den_names`: Retrieves a list of existing den sprites from the Sprite platform.
  +    *   `_choose_den_name`: Prompts the user to select a den name from a list of available ones.
  +    *   `_sprite_exists`: Checks if a specific den sprite exists.
  +    *   `_create_sprite`: Creates a new sprite on the platform.
  +    *   `_should_bundle_path`: Determines if a file/directory should be included in the sprite bundle during synchronization.
  +    *   `_sync_repo_to_sprite`: Archives and uploads a local repository snapshot to a remote sprite, returning the remote path.
  +    *   `_sesame_command`: Resolves the path to the `sesame` executable, handling potential `FileNotFoundError`.
  +    *   `_owned_porkbun_domains`: Fetches a list of domains owned via Porkbun, either from `sesame` CLI output or configuration files.
  +    *   `_sprite_url`: Retrieves the public URL for a given den sprite.
  +    *   `_attach_custom_domain`: Configures a custom domain for a den by making the sprite URL public and creating a URL forward using `sesame`.
  +
  +*   **CLI Commands**:
  +    *   `setup`: The core command for inferring a den's configuration from a repository, generating a `den.dhall` file, and running a Dhall generator script. It supports `--force` for overwriting and `--print` for dry-run output.
  +    *   `exec`: Executes a command within a specified den. (`test_cli_commands.py` indicates its use)
  +    *   `sprite-use` (aliased as `use`): Binds the current directory to a den. (`test_cli_commands.py` indicates its use)
  +    *   `logs`: Lists sessions or attaches to logs for a den. (`test_cli_commands.py` indicates its use)
  +    *   `redeploy`: Creates a checkpoint, finds its version, and restores it, effectively performing a redeploy. (`test_cli_commands.py` indicates its use)
  +
  +### 3.3. `den_cli.mcp_server` (Multi-Agent Control Protocol Server)
  +
  +The `den_cli.mcp_server` module provides an API-first interface for den operations using `fastmcp`, facilitating integration with automated systems.
  +
  +*   **Server Initialization**:
  +    *   `mcp`: An instance of `FastMCP` configured as "den-mcp", providing high-level instructions for its use.
  +    *   `PROJECT_DIR`: Defines the expected base directory for den projects.
  +
  +*   **Execution Primitives**:
  +    *   `StepResult`: A `TypedDict` defining the structured output for each command execution step, including `command`, `ok` status, `exit_code`, `stdout`, and `stderr`.
  +    *   `WorkflowError`: A `TypedDict` for standardized error reporting, detailing the `kind` of error, `message`, `failing_step`, and `remediation` advice.
  +    *   `_run_step`: The core function for executing shell commands within the MCP server. It encapsulates `subprocess.run` with robust error handling, timeout management, and comprehensive logging into a `StepResult`.
  +    *   `_command_exists`: Checks for command existence, leveraging `_run_step`.
  +    *   `_sesame_command`: Resolves the `sesame` executable path.
  +    *   `_build_error`: Constructs a `WorkflowError` from a `StepResult` and custom messages.
  +    *   `_result`: A helper to format the final structured workflow result, including overall `ok` status, `data`, `error` (if any), `next_steps`, and a list of all `StepResult` objects.
  +
  +*   **Domain Management Helpers**:
  +    *   `_sesame_owned_domains`: Retrieves a list of owned domains using `sesame`, falling back to config file discovery.
  +    *   `_sesame_url_forward_command`: Constructs the `sesame` command to add a URL forward for a custom domain.
  +
  +*   **MCP Tools (API Endpoints)**:
  +    *   `provision_den`: A comprehensive workflow tool for setting up a new den, including prerequisite checks (e.g., `sprite` and `sesame` CLI presence, `PROJECT_DIR` existence), Sprite authentication, sprite creation, and optional custom domain attachment. It returns a structured result, detailing all steps and any encountered errors.
  +    *   `operate_den`: (Implied from `test_mcp_server.py`) Handles various lifecycle actions for existing dens, such as `list`, `destroy`, `status`, and `domain` management. It performs necessary safety checks and executes corresponding `sprite` or `sesame` commands.
  +    *   `diagnose_den`: (Implied from `test_mcp_server.py`) Runs a series of diagnostic checks (e.g., `mypy`, `property_tests`, `den_smoke` and optionally `den_full` with Docker build) to assess the health and correctness of the den environment and related tools. It stops on the first failure.
  +
  +## 4. Data Flow
  +
  +This section details how information moves through the `den_cli` system, from user input to external tool interactions and structured outputs.
  +
  +### 4.1. Command Execution Flow
  +
  +1.  **Input**:
  +    *   **CLI**: A user invokes `den <command> [args]` (e.g., `den exec myproject echo hello`). This input is parsed by `Typer` (L29-L34 in `src/den_cli/cli.py`).
  +    *   **MCP Server**: An external system makes an API call to an MCP tool (e.g., `provision_den`, `operate_den`, `diagnose_den`) (L158 in `src/den_cli/mcp_server.py`).
  +
  +2.  **Command Construction**:
  +    *   Both CLI and MCP components utilize functions from `den_cli.core` to build the actual shell commands. For example, `sprite_exec_command` (L87-L90 in `src/den_cli/core.py`) is used by `cli.exec` to create the `sprite exec` argv. `normalize_den_name` (L55-L58 in `src/den_cli/core.py`) is frequently called to ensure correct den naming.
  +
  +3.  **Execution**:
  +    *   **CLI**: Commands are executed via `_run_checked` (L60-L75 in `src/den_cli/cli.py`), which wraps `subprocess.run` and includes error handling.
  +    *   **MCP Server**: Commands are executed via `_run_step` (L45-L82 in `src/den_cli/mcp_server.py`), providing detailed `StepResult` (L21-L29 in `src/den_cli/mcp_server.py`) for each command, including timing, stdout, and stderr.
  +
  +4.  **Output & Error Reporting**:
  +    *   **CLI**: Raw `stdout`/`stderr` are printed to the console. `CommandError` (L43-L44 in `src/den_cli/cli.py`) is raised on non-zero exit codes.
  +    *   **MCP Server**: A structured JSON response is returned, containing an overall `ok` status, `data`, a list of `StepResult` objects, and a detailed `WorkflowError` (L32-L42 in `src/den_cli/mcp_server.py`) on failure, constructed by `_build_error` (L97-L108 in `src/den_cli/mcp_server.py`).
  +
  +### 4.2. Den Setup and Provisioning Flow
  +
  +1.  **Repository Scan**:
  +    *   The `setup` command (L222 in `src/den_cli/cli.py`) or `provision_den` tool (L158 in `src/den_cli/mcp_server.py`) is invoked with a repository path.
  +    *   `detect_project_markers` (L135-L151 in `src/den_cli/core.py`) scans this path to identify configuration files like `package.json`, `pyproject.toml`, `Dockerfile`, etc., populating a `ProjectMarkers` (L20-L34 in `src/den_cli/core.py`) object.
  +
  +2.  **Setup Inference**:
  +    *   `infer_den_setup` (L154-L185 in `src/den_cli/core.py`) takes the `ProjectMarkers` and the repository name to infer the optimal `InferredDenSetup` (L37-L46 in `src/den_cli/core.py`). This includes choosing a `backend` (`nix` or `guix`) via `_infer_backend` (L188-L224 in `src/den_cli/core.py`), and a set of `nix_packages` (L227-L243 in `src/den_cli/core.py`) or `guix_packages` (L246-L260 in `src/den_cli/core.py`).
  +
  +3.  **Dhall Configuration Generation**:
  +    *   The `InferredDenSetup` is passed to `render_den_dhall` (L263 in `src/den_cli/core.py`), which produces a declarative Dhall configuration string.
  +
  +4.  **Artifact Materialization**:
  +    *   For the CLI `setup` command, if not in `--print-only` mode, the generated Dhall string is written to `den.dhall` (L275-L278 in `src/den_cli/cli.py`).
  +    *   Then, a `generate-from-dhall.sh` script (L279-L281 in `src/den_cli/cli.py`) is executed to create backend-specific files (e.g., Nix flakes or Guix manifests) based on the `den.dhall` configuration.
  +
  +5.  **Sprite Creation (Provisioning)**:
  +    *   During `provision_den`, after preflight checks (e.g., `_command_exists` for `sprite` (L174-L177 in `src/den_cli/mcp_server.py`)), the `sprite create` command (L220-L221 in `src/den_cli/mcp_server.py`) is executed, often with the normalized den name.
  +
  +### 4.3. Domain Management Flow
  +
  +1.  **Domain Information Gathering**:
  +    *   When a custom domain is specified (e.g., in `provision_den` or `operate_den domain`), `_sesame_owned_domains` (L123-L140 in `src/den_cli/mcp_server.py` and L172-L188 in `src/den_cli/cli.py`) is called to retrieve a list of domains owned by the user, primarily by invoking `sesame domain list --json` (L123-L140 in `src/den_cli/mcp_server.py`). It falls back to `discover_porkbun_domains_from_sesame_config` if `sesame` output is unavailable or malformed.
  +    *   The public URL of the den sprite is obtained using `_sprite_url` (L191-L197 in `src/den_cli/cli.py`), which parses the output of `sprite url`.
  +
  +2.  **Domain Processing**:
  +    *   `split_custom_domain` (L204 in `src/den_cli/cli.py` and L146 in `src/den_cli/mcp_server.py`) takes the desired custom domain and the list of owned domains to identify the correct zone and any subdomain.
  +
  +3.  **URL Forwarding**:
  +    *   The sprite's URL is first updated to be public (`sprite url update --auth public`) (L200-L203 in `src/den_cli/cli.py`).
  +    *   Then, `sesame` is used to create a permanent URL forward. The command is constructed using `_sesame_url_forward_command` (L143-L155 in `src/den_cli/mcp_server.py`) and executed to point the custom domain to the sprite's public URL.
  +
  +## 5. Testing Strategy
  +
  +The project employs comprehensive testing:
  +*   **Property-based testing (`hypothesis`, `antithesis`)**: Used in `test_core_properties.py` and `test_antithesis/test_core_properties.py` to verify invariants of core functions like `normalize_den_name`, `short_den_name`, `split_custom_domain`, and setup inference across a wide range of inputs.
  +*   **Unit/Integration testing (`unittest.mock`, `pytest`)**:
  +    *   `test_mcp_server.py` tests the `mcp_server`'s tools by mocking `_run_step` to simulate command successes and failures, ensuring structured outputs and error handling are correct.
  +    *   `test_cli_commands.py` tests the CLI commands by mocking `_run_checked` and `_run` to verify that the correct `sprite` or `sesame` commands are constructed and executed, and that user prompts work as expected. These tests often use `CliRunner` for invoking CLI commands.

  +# Comprehensive Guide to `den-cli`
  +
  +This document provides a detailed, step-by-step tutorial on understanding and utilizing the `den-cli` codebase. `den-cli` is a command-line interface and a Machine Comprehension Program (MCP) server designed to streamline the management of remote development environments, referred to as "dens," built on Sprite/Fly.io and integrated with domain services like Porkbun via `sesame`.
  +
  +## 1. Core Concepts and Utilities (`den_cli.core`)
  +
  +The `den_cli.core` module serves as the foundational layer, providing essential utilities for naming conventions, project detection, backend inference, Dhall configuration generation, and Sprite CLI command construction.
  +
  +### 1.1 Den Name Normalization
  +
  +`den-cli` enforces a specific naming convention for "dens" to ensure consistency with the underlying Sprite platform.
  +
  +*   The `normalize_den_name` function ensures that any user-provided name is prefixed with "den-".
  +    ```python
  +    return name if name.startswith("den-") else f"den-{name}"
  +    ```
  +*   Conversely, the `short_den_name` function removes this prefix, often used for display purposes.
  +    ```python
  +    return name[4:] if name.startswith("den-") else name
  +    ```
  +*   These functions are crucial for maintaining the correct naming format throughout the CLI and server operations, as seen in various tests. For example, `test_normalize_den_name_is_idempotent` verifies the idempotency of normalization.
  +    ```python
  +    normalized = normalize_den_name(name)
  +    assert normalized.startswith("den-")
  +    assert normalize_den_name(normalized) == normalized
  +    ```
  +    And `test_short_name_round_trip` ensures round-trip consistency.
  +    ```python
  +    normalized = normalize_den_name(name)
  +    short = short_den_name(normalized)
  +    assert normalize_den_name(short) == normalized
  +    ```
  +
  +### 1.2 Project Marker Detection
  +
  +To intelligently configure a den, `den-cli` first analyzes the local project directory for specific files that indicate the project's technology stack.
  +
  +*   The `detect_project_markers` function inspects the given `root` path for the presence of files like `package.json`, `pyproject.toml`, `Cargo.toml`, `Dockerfile`, `flake.nix`, and Guix-related files.
  +    ```python
  +    return ProjectMarkers(
  +        has_package_json=(root / "package.json").is_file(),
  +        has_bun_lock=(root / "bun.lock").is_file() or (root / "bun.lockb").is_file(),
  +        # ... and so on for other markers
  +    )
  +    ```
  +*   This detection logic is thoroughly tested by `test_detect_project_markers_matches_filesystem`, which creates various combinations of marker files in a temporary directory and asserts that `detect_project_markers` correctly identifies them.
  +    ```python
  +    if has_package_json:
  +        (tmp_path / "package.json").write_text("{}")
  +    # ... create other files
  +    markers = detect_project_markers(tmp_path)
  +    assert markers.has_package_json is has_package_json
  +    # ... assert other markers
  +    ```
  +
  +### 1.3 Backend Inference
  +
  +Once project markers are detected, `den-cli` infers the most suitable backend (Nix or Guix) and a set of default packages and environment variables for the den.
  +
  +*   The `infer_den_setup` function orchestrates this process, first calling `detect_project_markers` and then `_infer_backend`.
  +    ```python
  +    markers = detect_project_markers(root)
  +    backend, reasons = _infer_backend(markers)
  +    # ... further processing
  +    ```
  +*   The `_infer_backend` function implements the core logic for choosing between "nix" and "guix" based on a hierarchy of signals. For instance, explicit Guix manifests take precedence. If no Nix signals are present but a `pyproject.toml` exists, Guix is preferred. Otherwise, Nix is the default.
  +    ```python
  +    if markers.has_guix_manifest or markers.has_guix_channels:
  +        reasons.append("existing Guix manifests detected")
  +        return "guix", tuple(reasons)
  +    # ... more logic for Nix and Guix preference
  +    return "nix", tuple(reasons)
  +    ```
  +*   Helper functions `_infer_nix_packages` and `_infer_guix_packages` then determine a sensible default set of packages for the chosen backend based on the same markers.
  +    ```python
  +    packages = ["fish", "git", "helix"]
  +    if markers.has_bun_lock:
  +        packages.append("bun")
  +    # ... add other packages
  +    return tuple(dict.fromkeys(packages))
  +    ```
  +*   Tests like `test_infer_den_setup_prefers_guix_for_standalone_pyproject` confirm these inference rules.
  +    ```python
  +    (root / "pyproject.toml").write_text("[project]\nname='x'\n")
  +
  +    inferred = infer_den_setup(root)
  +
  +    assert inferred.backend == "guix"
  +    ```
  +    Antithesis property tests also ensure robust inference.
  +    ```python
  +    if markers.has_pyproject and not (
  +        markers.has_bun_lock
  +        or markers.has_package_json
  +        or markers.has_mise_toml
  +        or markers.has_flox_toml
  +        or markers.has_nix_flake
  +        or markers.has_shell_nix
  +        or markers.has_dockerfile
  +        or markers.has_containerfile
  +        or markers.has_helm_chart
  +    ):
  +        sometimes(
  +            inferred.backend == "guix",
  +            "standalone pyproject repositories can infer guix backend",
  +            {"repo_name": repo_name, "backend": inferred.backend},
  +        )
  +    ```
  +
  +### 1.4 Dhall Configuration Rendering
  +
  +The inferred den setup is then translated into a Dhall configuration file (`den.dhall`), which describes the den's properties in a structured, reproducible format.
  +
  +*   The `render_den_dhall` function takes an `InferredDenSetup` object and the path to the Dhall templates (`dhall_dir`) to produce the final Dhall configuration string.
  +    ```python
  +    backend_expr = "Types.Backend.Nix" if config.backend == "nix" else "Types.Backend.Guix"
  +    dockerfile_expr = _optional_text(config.dockerfile)
  +    nix_expr = _render_nix_config(config.nix_packages) if conf... (truncated)
  +    ```
  +*   The `test_render_den_dhall_is_self_consistent` test verifies that the generated Dhall string contains expected values based on the input configuration.
  +    ```python
  +    rendered = render_den_dhall(config, Path("/tmp/dhall"))
  +
  +    assert f'name = "{name}"' in rendered
  +    assert f"backend = Types.Backend.{backend.capitalize()}" in rendered
  +    ```
  +
  +### 1.5 Sprite Command Generation
  +
  +`den-cli` extensively interacts with the Sprite CLI. A set of helper functions is provided to construct these commands consistently.
  +
  +*   The `sprite_command` function is a generic builder for Sprite commands, handling optional organization (`-o`) and normalized sprite name (`-s`).
  +    ```python
  +    command = ["sprite"]
  +    org = sprite_org()
  +    if org:
  +        command.extend(["-o", org])
  +    if sprite_name:
  +        command.extend(["-s", normalize_den_name(sprite_name)])
  +    command.extend(args)
  +    return command
  +    ```
  +*   Specialized functions exist for common operations:
  +    *   `sprite_exec_command`: Executes a non-interactive command inside a den.
  +        ```python
  +        return sprite_command("exec", "--", *command, sprite_name=name)
  +        ```
  +    *   `sprite_tty_exec_command`: Executes an interactive TTY session.
  +        ```python
  +        return sprite_command("exec", "--tty", "--", *command, sprite_name=name)
  +        ```
  +    *   `sprite_use_command`: Binds a local directory to a den.
  +        ```python
  +        command = ["sprite"]
  +        org = sprite_org()
  +        if org:
  +            command.extend(["-o", org])
  +        command.extend(["use", normalize_den_name(name)])
  +        return command
  +        ```
  +    *   `sprite_logs_command`: Lists sessions or attaches to a den's logs.
  +        ```python
  +        if list_only:
  +            return sprite_command("sessions", "list", sprite_name=name)
  +        if selector:
  +            return sprite_command("attach", selector, sprite_name=name)
  +        return sprite_command("attach", sprite_name=name)
  +        ```
  +*   These command builders are tested to ensure they produce the correct `argv` lists, as seen in `test_sprite_command_uses_normalized_name`, `test_sprite_exec_command_preserves_payload`, etc.
  +    ```python
  +    command = sprite_exec_command(name, payload)
  +    assert command[:5] == ["sprite", "-s", normalize_den_name(name), "exec", "--"]
  +    assert command[5:] == payload
  +    ```
  +
  +### 1.6 Checkpoint Management
  +
  +For operations like redeploying a den, `den-cli` utilizes Sprite's checkpointing capabilities.
  +
  +*   `make_sprite_redeploy_comment` generates a unique comment for checkpoints, allowing them to be identified later.
  +    ```python
  +    return f"den-redeploy:{normalize_den_name(name)}:{nonce}"
  +    ```
  +*   `find_checkpoint_version_in_api_output` and `find_checkpoint_version_in_list_output` parse Sprite CLI output to extract the version ID of a specific checkpoint.
  +    ```python
  +    payload = json.dumps([{"id": version_id, "comment": comment}])
  +    assert find_checkpoint_version_in_api_output(payload, comment) == version_id
  +    ```
  +*   `sprite_checkpoint_create_command` and `sprite_restore_command` generate the respective Sprite commands.
  +    ```python
  +    return sprite_command("checkpoint", "create", "--comment", comment, sprite_name=name)
  +    ```

  +### 1.7 Custom Domain Handling
  +
  +`den-cli` supports custom domains by integrating with `sesame` (a domain management tool) and Porkbun.
  +
  +*   `split_custom_domain` helps determine the appropriate domain zone and subdomain for a given custom domain, considering a list of owned domains.
  +    ```python
  +    for owned in sorted(owned_domains, key=len, reverse=True):
  +        if host.endswith(owned):
  +            zone = owned
  +            subdomain = host[: -len(owned)].rstrip(".")
  +            break
  +    # ... return zone, subdomain
  +    ```

  +    ```python

  +## 2. Command-Line Interface (`den_cli.cli`)
  +
  +The `den_cli.cli` module implements the user-facing command-line interface using `typer`. It orchestrates calls to the `core` module and external commands.
  +
  +### 2.1 Core CLI Utilities
  +
  +Several helper functions manage command execution, project directory checks, and interactive user input.
  +
  +*   `_run` executes a shell command without `check=True`, meaning it won't raise an exception on non-zero exit codes.
  +    ```python
  +    return subprocess.run(
  +        cmd,
  +        cwd=str(cwd) if cwd else None,
  +        text=True,
  +        input=input_text,
  +        capture_output=capture_output,
  +        check=False,
  +    )
  +    ```
  +*   `_run_checked` wraps `_run`, raising a `CommandError` if the command fails, and prints relevant output to the console.
  +    ```python
  +    if proc.returncode != 0:
  +        if error_hint:
  +            typer.secho(error_hint, fg=typer.colors.RED)
  +        if proc.stderr:
  +            typer.echo(proc.stderr.strip())
  +        if proc.stdout and not proc.stderr:
  +            typer.echo(proc.stdout.strip())
  +        raise CommandError(f"command failed: {' '.join(cmd)}")
  +    return proc
  +    ```
  +*   `_command_exists` checks if a given command is available in the system's PATH.
  +    ```python
  +    return _run(["bash", "-lc", f"command -v {name}"], capture_output=True).returncode == 0
  +    ```
  +*   `_ensure_project_dir` verifies that the `PROJECT_DIR` exists.
  +    ```python
  +    if not PROJECT_DIR.is_dir():
  +        raise CommandError(f"den project not found at {PROJECT_DIR}")
  +    ```
  +*   `_list_den_names` fetches a list of existing dens from Sprite.
  +    ```python
  +    proc = _run_checked(sprite_command("list", "-prefix", "den-"), capture_output=True)
  +    names = [line.strip() for line in proc.stdout.splitlines() if line.strip()]
  +    return [name for name in names if name.startswith("den-")]
  +    ```
  +*   `_choose_den_name` provides an interactive prompt for selecting a den from the available list.
  +    ```python
  +    selected = typer.prompt(prompt, default=names[0])
  +    if not selected.strip():
  +        raise CommandError("Cancelled")
  +    return normalize_den_name(selected.strip())
  +    ```

  +### 2.2 `den setup` Command
  +
  +The `setup` command is central to initializing a den for a new or existing repository.
  +
  +*   It infers the den's configuration using `infer_den_setup`, detects project markers with `detect_project_markers`, and renders the Dhall configuration using `render_den_dhall`.
  +    ```python
  +    inferred = infer_den_setup(repo_dir)
  +    markers = detect_project_markers(repo_dir)
  +    rendered = render_den_dhall(inferred, DHALL_DIR)
  +    ```
  +*   If `print_only` is specified, it just prints the Dhall config without writing files. Otherwise, it writes `den.dhall` and then executes a generator script (`generate-from-dhall.sh`) to materialize backend-specific files.
  +    ```python

  +        typer.echo(rendered)
  +    else:
  +        # ... write den.dhall
  +        _run_checked(
  +            ["bash", str(PROJECT_DIR / "scripts" / "generate-from-dhall.sh"), str(dhall_file), str(repo_dir)],

  +## 3. Machine Comprehension Program (MCP) Server (`den_cli.mcp_server`)
  +
  +The `den_cli.mcp_server` module exposes `den-cli` functionality as a `FastMCP` server, allowing AI agents or other automated systems to interact with and manage dens programmatically.
  +
  +### 3.1 Step Execution and Error Handling
  +
  +The server executes shell commands as discrete "steps" and provides structured results and error reporting.
  +
  +*   The `_run_step` function is the workhorse, executing a shell command (`subprocess.run`), capturing its output, exit code, and duration, and returning a `StepResult` dictionary. It handles timeouts gracefully.
  +    ```python
  +    proc = subprocess.run(
  +        command,
  +        cwd=run_cwd,
  +        text=True,
  +        input=input_text,
  +        capture_output=True,
  +        check=False,
  +        timeout=timeout_s,
  +    )
  +    # ... construct StepResult
  +    ```
  +*   `_build_error` transforms a `StepResult` into a structured `WorkflowError` dictionary, including remediation advice, for standardized error reporting.
  +    ```python
  +    return {
  +        "kind": "command_failure",
  +        "message": message,
  +        "failing_step": step["step"],
  +        "command": step["command"],
  +        "exit_code": step["exit_code"],
  +        "timed_out": step["timed_out"],
  +        "stdout": step["stdout"],
  +        "stderr": step["stderr"],
  +        "remediation": remediation,
  +    }
  +    ```
  +*   `_result` creates a standardized workflow result dictionary, including whether the operation was successful, any data, errors, and a list of executed steps.
  +    ```python
  +    return {
  +        "workflow": workflow,
  +        "ok": ok,
  +        "data": data or {},
  +        "error": error,
  +        "next_steps": next_steps or [],
  +        "steps": steps,
  +    }
  +    ```
  +*   Tests in `test_mcp_server.py` extensively use `_fake_run_step` and `_fake_run_step_fail` to simulate command execution and verify the error and result structures.
  +    ```python
  +    def _fake_run_step(
  +        step: str,
  +        command: list[str],
  +        *,
  +        cwd: Path | None = None,
  +        timeout_s: int = 120,
  +        input_text: str | None = None,
  +    ) -> mcp_server.StepResult:
  +        """Simulate a successful step."""
  +        return {
  +            "step": step,
  +            "command": command,
  +            "cwd": str(cwd or Path.home()),
  +            "ok": True,
  +            "exit_code": 0,
  +            "timed_out": False,
  +            "duration_ms": 10,
  +            "stdout": "",
  +            "stderr": "",
  +        }
  +    ```
  +
  +### 3.2 Prerequisites and Dependencies
  +
  +The MCP server includes checks for external dependencies like `sprite` and `sesame`.
  +
  +*   `_command_exists` uses `_run_step` to check for command availability.
  +    ```python
  +    result = _run_step("check_command", ["bash", "-lc", f"command -v {cmd}"], cwd=None, timeout_s=15)
  +    return result["ok"]
  +    ```
  +*   `_sesame_command` resolves the path to the `sesame` executable.
  +    ```python
  +    return resolve_sesame_command()
  +    ```
  +*   `_sesame_owned_domains` uses `sesame` to list owned domains, falling back to configuration-based discovery if the command fails.
  +    ```python
  +    step = _run_step("sesame_domain_list", _sesame_command() + ["domain", "list", "--all", "--json"], cwd=None, timeout_s=30)
  +    if not step["ok"] or not step["stdout"].strip():
  +        return discover_porkbun_domains_from_sesame_config()
  +    try:
  +        payload = json.loads(step["stdout"])
  +    except json.JSONDecodeError:
  +        return discover_porkbun_domains_from_sesame_config()
  +    if not isinstance(payload, list):
  +        return discover_porkbun_domains_from_sesame_config()
  +
  +    domains: list[str] = []
  +    for row in payload:
  +        if not isinstance(row, dict):
  +            continue
  +        domain = row.get("domain")
  +        if isinstance(domain, str) and domain:
  +            domains.append(domain)
  +    return domains or discover_porkbun_domains_from_sesame_config()
  +    ```
  +*   `test_provision_den_fails_without_sprite` demonstrates how missing dependencies are handled.
  +    ```python
  +    with patch.object(mcp_server, "_command_exists", return_value=False):
  +        result = mcp_server.provision_den(name="myproject")
  +
  +    assert result["ok"] is False
  +    assert result["error"]["kind"] == "missing_dependency"
  +    ```

  +### 3.3 `provision_den` Tool
  +
  +This MCP tool encapsulates the entire workflow for creating and setting up a new den, including optional custom domain configuration.
  +
  +*   It first checks for `sprite` and `sesame` dependencies.
  +    ```python
  +    for cmd in ("sprite",):
  +        exists = _command_exists(cmd)
  +        if not exists:
  +            return _result(
  +                "provision_den",
  +                False,
  +                steps=steps,
  +                error={
  +                    "kind": "missing_dependency",
  +                    "message": f"Required command not found: {cmd}",
  +                    "failing_step": "preflight",
  +                    "command": ["command", "-v", cmd],
  +                    "exit_code": None,
  +                    "timed_out": False,
  +                    "stdout": "",
  +                    "stderr": f"{cmd} is not on PATH",
  +                    "remediation": [
  +                        "Install the Sprite CLI and authenticate with Fly.",
  +                        "Retry provision_den after sprite list succeeds.",
  +                    ],
  +                },

  +*   It verifies the `PROJECT_DIR` exists.
  +    ```python
  +    if not PROJECT_DIR.is_dir():
  +        return _result(
  +            "provision_den",
  +            False,
  +            steps=steps,
  +            error={
  +                "kind": "missing_project",
  +                "message": f"den project directory not found: {PROJECT_DIR}",
  +                "failing_step": "preflight",
  +                "command": ["test", "-d", str(PROJECT_DIR)],
  +                "exit_code": None,
  +                "timed_out": False,

  +                "stderr": "Project directory missing",

  +*   It then runs a series of `_run_step` calls for Sprite authentication checks, Sprite creation, and optionally, making the Sprite URL public and attaching a custom domain using `_sesame_url_forward_command`.

  +                "provision_den",
  +                False,
  +                steps=steps,
  +                error=_build_error(
  +                    step,
  +                    f"Provisioning failed at {step_name}",

  +                        "Inspect stderr/stdout in this error payload.",

  +                    ],
  +                ),
  +            )

  +### 3.5 `diagnose_den` Tool

  +This tool runs a series of diagnostic checks to verify the health and setup of the `den-cli` environment.

  +## 4. Testing Strategy
  +
  +The codebase utilizes several testing frameworks to ensure reliability and correctness.
  +
  +*   **`pytest`**: The standard testing framework for Python, used for most unit and integration tests.
  +*   **`hypothesis`**: A property-based testing library, heavily used in `test_core_properties.py` to generate diverse inputs and test invariants of functions like `normalize_den_name`, `short_den_name`, and `parse_sprite_url`. This helps catch edge cases that might be missed by example-based tests.
  +    ```python
  +    @given(SAFE_NAME)
  +    def test_normalize_den_name_is_idempotent(name: str) -> None:
  +        normalized = normalize_den_name(name)
  +        assert normalized.startswith("den-")
  +        assert normalize_den_name(normalized) == normalized
  +    ```
  +*   **`antithesis`**: An advanced property-based testing and fuzzing tool, used in `tests/antithesis/test_core_properties.py` to find deeper bugs and verify long-running invariants, especially for stateful transitions or complex interactions. It uses `always`, `sometimes`, and `reachable` assertions.
  +    ```python
  +    always(
  +        normalized.startswith("den-"),
  +        "normalize_den_name adds den prefix",
  +        {"name": name, "normalized": normalized},
  +    )
  +    # ...
  +    sometimes(
  +        inferred.backend == "guix",
  +        "adding guix metadata can force guix backend inference",
  +        {"repo_name": repo_name, "states": states, "backend": inferred.backend},

###

Your task is to help verify the quality of the diff.
We care only about specific categories of important issues.
The rubric below outlines these categories of important issues, and contains guidelines and examples to correctly identify them:

---
**repetitive_or_duplicate_code**:
Guidelines:
    Repetitive or duplicate code.
Examples:
    - A non-trivial calculation or piece of logic is repeated in multiple places within a file.
    - New code is introduced by the diff to accomplish a certain functionality, but there is an existing function in the code base that already implements the same functionality, or could be easily generalized to accomplish the desired functionality.
    - A file is duplicated (make exceptions for cases where duplication may be necessary such as test files).
    - A significant amount of code is introduced which duplicates functionality from standard or well-known libraries.
    - Multiple functions format or build the same string or data structure in the same way without using a shared helper function or test fixture.
Exceptions:
    - Do not flag duplication between legacy and new implementations when the codebase is clearly undergoing a migration or maintaining multiple versions for compatibility.
    - Do not flag duplication across different architectural layers or modules when the duplication serves to maintain proper separation of concerns.

---
**incorrect_algorithm**:
Guidelines:
    - Code that implements an algorithm incorrectly for its stated purpose.
    - Look for: sorting algorithms with wrong comparison logic, search algorithms with incorrect termination, mathematical calculations with wrong formulas, data structure operations that don't maintain invariants, algorithms that don't handle edge cases (empty inputs, single elements).
    - Only flag issues that are clearly incorrect for the stated purpose of the algorithm, and describe the problem and correction in detail.

---
**insecure_code**:
Guidelines:
    - Look for hard-coded secrets such as API keys, passwords, tokens, or credentials in the diff.
    - Check for variable names containing: 'token', 'key', 'secret', 'password', 'credential', 'auth'
    - Look for string literals that appear to be:
        - API keys or tokens (long alphanumeric strings, often 20+ characters)
        - Hexadecimal strings that could be tokens or keys
        - URLs with embedded credentials (e.g., 'https://user:password@host')
        - Connection strings with passwords
    - Flag any credentials or secrets that should be loaded from environment variables or configuration files instead.
Examples:
    - A variable named `api_key` or `auth_token` is assigned a hard-coded string value.
    - A connection string contains a hard-coded password.
    - A long hexadecimal string is assigned to a variable with 'token' in its name.
    - An API request includes a hard-coded authentication header value.

---
**type_safety_violation**:
Guidelines:
    - Code that violates type safety expectations or could cause type-related runtime errors.
    - Look for: incorrect type assumptions, missing type checks before operations, unsafe type casting, attribute access on potentially None values.
    - Check for return type violations: functions that return values inconsistent with their declared return type annotations (e.g., returning None when the type annotation specifies a non-optional type, returning wrong tuple element types).

---
**fails_silently**:
Guidelines:
    Code that fails silently is code that ignores errors without reporting them.
Examples:
    - Overly broad exception handlers (e.g. bare 'except' or 'except Exception') that catch errors and continue execution without handling, logging, or re-raising them.
    - The return value of a function that returns an error value in case of a failure is not checked by the caller.
Exceptions:
    - There are certain cases where broad exception handlers are acceptable, such as in an executor class or in a main loop that iterates over several tasks. Such cases should still properly log and report the errors.
    - Do not raise issues related to potential program crashes.

---
**user_request_artifacts_left_in_code**:
Guidelines:
    - Comments should describe what the code does, not how it was changed.
    - Flag comments that reference the change process: '# Changed from X to Y', '# Updated to print less'
    - Flag comments that mention fixing or addressing issues: '# Fixed bug where...', '# This addresses...'
    - Flag documentation written in past tense about modifications
    - Acceptable comments explain current behavior without referencing changes:
        - Acceptable: '# Multiply by 3x' instead of '# Reduced factor from 5x to 3x'
        - Acceptable: '# Handle edge case' instead of '# Fixed edge case bug'

---
**error_handling_missing**:
Guidelines:
    - Missing error handling for operations that could reasonably fail.
    - Look for: file I/O without exception handling, network requests without timeout/retry logic, user input processing without validation, external API calls without error checking, database operations without transaction handling.
    - Only flag issues that are clearly incorrect, and avoid flagging issues where it is not a big problem (e.g. file I/O in a script may not need flagging while missing error handling for file I/O in a long running or production systems should have error handling).

---
**logic_error**:
Guidelines:
    - Logic errors are flaws in the reasoning or flow of the code that would cause incorrect behavior.
    - Look for: off-by-one errors in loops or array indexing, incorrect conditional logic (wrong operators, inverted conditions), variable assignments that overwrite needed values, incorrect order of operations, missing or incorrect loop termination conditions, algorithms that don't match their intended purpose.
    - Look for missing, incorrect, or incomplete parameters to function/API calls that will cause the function to behave differently than intended (e.g., missing event masks, wrong flags, omitted required options).
    - Pay special attention to control flow issues: early returns or breaks that prevent intended functionality from executing, functions that exit before completing their stated purpose, conditions that prevent code paths from being reached when they should be.
    - Do not flag issues that are not clearly incorrect, for example it's possible code is implemented in a suboptimal way, this is not an issue unless it is explicitly stated that the code should be optimal or implemented in a certain way.

---
**refactoring_needed**:
Guidelines:
    - Functions that have gotten long (> 50 lines) and are mixing multiple concerns and/or combining several different steps should be broken up. (Typically by using helper functions and/or separate classes to encapsulate individual concerns.)
    - Classes or files that are combining different concerns should be broken up, such that each class / file only deals with one primary concern.
        - Note: we don't impose any minimal or maximal length on a class or file. Classes and files are ok to be long, as long as they only deal with a single concern.
Examples:
    - New functionality that is orthogonal to the existing functionality in a function is inserted into the existing function's body instead of being separated out into its own function.
    - A class mixes two different use cases that could be separated into two classes.

---
**commit_message_mismatch**:
Guidelines:
    - The diff must completely fulfill the user's request.
    - Look for incomplete implementations:
       - When the user asks for changes 'throughout', 'everywhere', or 'all', verify ALL instances are modified
       - If multiple changes are requested, ensure each one is fully implemented
       - Check that fixes are applied to all occurrences of a pattern, not just some
    - Look for scope mismatches:
       - Changes only in initialization when they should apply during execution
       - Modifications to only one file when multiple files need updates
       - Partial refactoring that leaves related code unchanged
    - Look for unauthorized changes:
       - Configuration changes (linting, build, test settings) not requested
       - New features or options beyond the request
       - Changes to unrelated code
       - Binaries, compiled files, dependencies, or build artifacts that should not be in version control
    - Look for unintended removals:
       - Removal of project-specific configuration or settings that should be preserved
       - Deletion of functionality that is still needed
       - Loss of necessary entries when replacing configuration files
    - Look for unintended side effects:
       - Changes that affect code paths or functionality beyond what was requested
       - Modifications that impact how existing features work in ways not mentioned in the request
Exceptions:
    - Minor refactors directly related to requested changes are acceptable.

---
**test_coverage**:
Guidelines:
    - If the diff introduces significant new functionality, and the code base has existing unit and/or integration tests, new tests should be added to cover the new functionality.
    - If the diff changes the behavior of existing functionality that is covered by automated tests, those tests should be updated to reflect the new behavior.
    - If the diff contains a bug fix, and the code base has existing unit and/or integration tests, a regression test should be added for the bug.
Exceptions:
    - Syntactical or logical issues in tests will be raised in other issue types and do not belong in this category.

---
**poor_naming**:
Guidelines:
    - File, class, function, function parameter, and constant names should follow the format and naming standards that are currently dominant in the code base (especially within the same file or folder), or the style guide if one exists.
    - In the absence of existing code, common naming standards for the given programming language should be used.
    - Function names should be descriptive of what the function does. A person reading the function name without seeing its implementation should be able to get a sense of its purpose.
    - If any function parameter is being mutated by the function, that fact should be made clear in the name of the parameter (e.g. by appending `_out` / `Out` to the name), unless it is clear that such name-based annotations would be against the existing naming style in the code base.
    - Note that we don't impose any specific criteria on the length of names. If the existing code base uses many abbreviated names, new code should follow that. Or if it uses a lot of long, verbose names, this similarly should be followed.
    - If a component's functionality is significantly changed, the name of the component should be updated to reflect the new functionality, if it is not already clear from the context.
Exceptions:
    - Short names for local variables are usually okay.

---
**abstraction_violation**:
Guidelines:
    - Code that breaks established abstraction boundaries within the codebase.
    - Look for:
        - Direct access to internal data structures of classes/modules that should be encapsulated
        - Bypassing public APIs to manipulate state or access internal functionality
        - Mixing of concerns that should be separated by layers or modules
        - Violating private vs. public interfaces (e.g., accessing private attributes or methods from outside their defining class/module)
Examples:
    - A Python function directly accesses a private attribute, variable or function (prefixed with an underscore) from a different class or file.
    - A module modifies the internal state of another module directly instead of using and/or adding public API functions.
Exceptions:
    - Unit tests that need to access internal state for verification purposes.

---
**instruction_file_disobeyed**:
Guidelines:
    Explicit instructions in files such as .claude.md, CLAUDE.md, and AGENTS.md MUST be obeyed.
Examples:
    - CLAUDE.md requests the use of single quotes only, but double quotes are used.
    - AGENTS.md requests that new versions be created on every database update, but a database entry is modified directly.
    - .claude.md says to always run the tests after making changes, but the agent did not run the tests.
Exceptions:
    - Instructions in the closest file _above_ a location take precedence. For example, when considering a file foo/bar.py, foo/CLAUDE.md takes precedence over CLAUDE.md.
    - Instructions only apply to the subtree below the file. For example, when considering a file foo/bar.py, foo/baz/CLAUDE.md does not apply.
    - Applicable instructions should ONLY be contravened in the case of explicit user request--but if the user does explicitly request something counter to the instruction files, this should not be reported as a disobeyed instruction file.

---
**async_correctness**:
Guidelines:
    - Issues specific to asynchronous or concurrent code correctness.
    - Look for: missing await keywords on functions that are clearly async (defined with 'async def' or returning coroutines/Promises), improper async context manager usage, race conditions in async code, deadlock potential, shared state access without proper synchronization.
    - For threading issues: threads or background tasks that are created but never started, joins without corresponding starts.
    - Be careful not to flag decorators or wrappers as requiring await unless you can verify they actually make the function async. Only flag clear cases where an async function is called without await or a thread/task is created but never started.

---
**documentation_implementation_mismatch**:
Guidelines:
    - The implementation should follow, in this priority order: 1. the user's request, 2. documentation existing in the code base, 3. existing code around it, 4. general best practices and common sense
    - If the user's request conflicts with the state of documentation in the code base, the documentation in the code base should be updated to reflect the new user request.
Examples:
    - The docstring of a class or function does not match what the class or function implements.
    - The repository contains a README.md file with instructions that are not adhered to by the code.
    - The diff implements significant new functionality, but existing documentation within the repository is not updated.
    - Inline comments are not updated even though functionality was changed by the diff.
    - Documentation contains outdated code snippets or commands that need to be updated because of the changes made by the diff.
Exceptions:
    - TODOs/FIXMEs that are not implemented yet are not considered a documentation mismatch.

---
**resource_leakage**:
Guidelines:
    - Focus on system resources that require explicit cleanup: file handles, network connections, database connections, memory allocations, and similar OS-level resources.
    - These resources must be reliably freed even if exceptions occur.
    - For these system resources, cleanup should use try/finally blocks, context managers (with statements), or RAII patterns.
    - Also look for reference management issues: objects being cleaned up while still holding references elsewhere, or cleanup operations (like garbage collection) called before removing all references to the object.
Examples:
    - A file or socket connection is opened but not reliably closed.
    - A database transaction is started but not committed or rolled back in all code paths.
    - Memory is allocated but not freed (in languages with manual memory management).
    - An object's cleanup method triggers garbage collection while the object is still referenced in a global data structure, preventing proper cleanup.
Exceptions:
    - Animation loops, timers, and intervals that are controlled by boolean flags or cleared by ID are not resource leaks if they have proper stop mechanisms.
    - Event listeners that are meant to persist for the lifetime of the application.
    - Resources that are automatically cleaned up by garbage collection (unless they hold system resources).

---
**incomplete_integration_with_existing_code**:
Guidelines:
    - The diff should follow existing architectural and organizational patterns in the codebase:
        - If the codebase uses a modular structure with separate files for classes/components, new classes should follow the same pattern
        - If the codebase organizes code in specific directories (e.g., src/, components/, utils/), new code should be placed accordingly
        - If the codebase uses specific import/export patterns (e.g., relative vs. absolute imports), new code should use the same patterns
    - The diff should integrate functionally with existing code by adding invocations, updating invocations, replacing code with newly defined functions or variables, removing duplicate code when a new piece replaces it, etc.
    - Prefer using existing library/dependency APIs over custom implementations when the library provides (or will provide) the needed functionality.
Examples:
    - The codebase uses absolute Python imports from the project root, but a new file uses relative imports.
    - The codebase places classes into separate files under a source directory, but new classes are all added to an existing file.
    - The diff implements a new function, but doesn't add any callsites for it
    - A new optional parameter is added to a function to implement a requested feature, but existing callsites are not updated to make use of this parameter
    - A named constant is introduced to replace a hard-coded inline literal, but existing code is not updated to make use of the new constant everywhere
    - Custom code is added to implement functionality that an existing external library already provides or will provide in newer versions.

---
**runtime_error_risk**:
Guidelines:
    - Code patterns that are very likely to cause runtime errors during execution.
    - Check for version compatibility issues: usage of function parameters, APIs, or language features that are only available in specific versions of the language, standard library, or external dependencies (e.g., a keyword argument added in Python 3.10 will cause TypeError on Python 3.8/3.9).
    - Look for: potential null/None pointer dereferences, array/list access with potentially invalid indices, division by zero possibilities, file operations without existence checks, network/IO operations without timeout or error handling, infinite loop conditions, memory allocation issues.
    - Check string encoding/decoding operations: calls to .encode() or .decode() without error handling (try/except or 'errors' parameter) that could raise UnicodeEncodeError or UnicodeDecodeError, especially when processing untrusted or streamed data.
    - Look for operations with global side effects that could cause problems: os.chdir() without proper restoration, modifying global state in ways that affect other code, operations that are not thread-safe when concurrency is present.
    - Only flag issues where there is clear evidence the code will fail or cause serious problems. Avoid speculating about potential issues in well-established language patterns or standard library usage.

---
**correctness_syntax_issues**:
Guidelines:
    - The diff should not contain any syntax errors that would prevent the code from running.
    - CAREFULLY CHECK INDENTATION: In Python and other indentation-sensitive languages, verify that all function definitions, class definitions, and code blocks maintain proper indentation levels. Dedenting a function body to the module level or similar indentation errors are critical syntax issues.
    - Look for: broken indentation that would cause syntax errors, missing or mismatched brackets/braces/parentheses, references to files/classes/functions that don't exist, removal of code that is still being referenced elsewhere.
    - Check function signatures match their usage: if a function is modified to return different values (e.g., a single value vs. a tuple), all call sites must be updated accordingly.
Examples:
    - The diff breaks the indentation of a Python function or class, dedenting it incorrectly.
    - Code references a file, class or function that does not exist, or removes a file, class or function that is definitely still being referenced.
    - A function is changed to return a tuple of two values, but existing callers still expect only a single return value.
    - A function's return statement is removed but callers still expect a return value.

---
**dependency_management**:
Guidelines:
    - Check all import statements in new or modified files. If new code imports a library or package that is not part of the language's standard library, verify that the dependency is listed in the repository's dependency/requirement files (e.g., requirements.txt, pyproject.toml, package.json, Gemfile, etc.).
    - If the diff removes the last remaining use of an external library or package, the dependency and/or requirement files in the repository should be updated to no longer include the library.
    - If the codebase uses a dependency for some functionality, the diff should avoid introducing other packages that provide the same functionality, unless there is a good reason to do so (e.g. the new package is significantly better maintained, has better performance, or is more secure).
Exceptions:
    - Do not raise issues related to package versions or pinning unless it is a critical issue.
